function displayPlaces(places) { // Clear the current content of the places list const placesList = document.getElementById('places-list'); placesList.innerHTML = ''; // Iterate over the places data places.forEach(place => { // For each place, create a div element and set its content const placeElement = document.createElement('div'); placeElement.className = 'place-card'; placeElement.setAttribute('data-price', place.price); placeElement.innerHTML = `

${place.name}

Prix : ${place.price}€/nuit

Lieu : ${place.location}

Voir les détails `; // Append the created element to the places list placesList.appendChild(placeElement); }); // Si aucun lieu n'est disponible, afficher un message if (places.length === 0) { placesList.innerHTML = '

Aucun lieu disponible pour le moment.

'; } }
